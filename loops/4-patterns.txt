#include<stdio.h>

int main() {
    int num = 5;
    for(int row = 1; row <= num; row++){
         for(int col = 1; col < row; col++){
            printf("  ");
        }
        for(int col = 1; col <= num; col++){
            if (row == 1 || row == num || col == 1 || col == num){
                printf("* ");
            }else{
                printf("  ");
            }
        }
        printf("\n");
    }
    return 0;
}

#include<stdio.h>

int main() {
    int num = 5;
    int flag = (num / 2) + 1;
    for(int row = 1; row <= num; row++){
        for(int col = 1; col <= num; col++){
            if (row == 1 || row == num || col == 1 || col == num ){
                printf("* ");
            }else{
                if (row == flag && col == flag){
                    printf("* ");
                }else{
                    printf("  ");
                }
            }
        }
        printf("\n");
    }
    return 0;
}
